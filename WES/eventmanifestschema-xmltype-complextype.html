XmlType Complex Type

Defines an XML fragment. Any schema instance is allowed; the top-level node in the fragment must contain a namespace attribute.

<xs:complexType name="XmlType">
    <xs:sequence>
        <xs:any
            processContents="skip"
            minOccurs="0"
            maxOccurs="unbounded"
            namespace="##other"
         />
    </xs:sequence>
</xs:complexType>
